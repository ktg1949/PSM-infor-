async function sbGet(table, filter='') {
  const r=await fetch(`${SB_URL}/rest/v1/${table}?${filter}&order=created_at.desc`,{method:'GET',headers:HEADERS});
  if(!r.ok){const t=await r.text();throw new Error(`${r.status}: ${t}`);}
  return r.json();
}
async function sbInsert(table, data) {
  const r=await fetch(`${SB_URL}/rest/v1/${table}`,{method:'POST',headers:{...HEADERS,'Prefer':'return=representation'},body:JSON.stringify(data)});
  if(!r.ok){const t=await r.text();throw new Error(`${r.status}: ${t}`);}
  return r.json();
}
async function sbDelete(table, id) {

}

async function doLogin(){
  const id=document.getElementById('loginId').value.trim();

  const err=document.getElementById('loginErr');
  const btn=document.getElementById('loginSubmit');
  err.style.display='none';

  if(localU){enterApp(localU);return;}
  btn.disabled=true;btn.textContent='로그인 중...';
  try{
    const rows=await sbGet('profiles',`username=eq.${encodeURIComponent(id)}&password=eq.${encodeURIComponent(pw)}`);
    if(!rows.length){err.textContent='아이디 또는 비밀번호가 올바르지 않습니다.';err.style.display='block';}

  finally{btn.disabled=false;btn.textContent='로그인';}
}

  document.getElementById('catSub').textContent=CATS[n].sub;
  render();
}

function updateBadges(){

  for(let i=1;i<=6;i++)document.getElementById('cnt'+i).textContent=allDocs.filter(d=>d.category==i).length;
  const f=curCat===0?allDocs:allDocs.filter(d=>d.category==curCat);
  document.getElementById('statTotal').textContent=allDocs.length;
  document.getElementById('statCur').textContent=f.length;
}

async function loadDocs(){
  const grid=document.getElementById('docGrid');
  grid.innerHTML=`<div class="loading-state"><div class="spinner"></div><p>문서를 불러오는 중...</p></div>`;
  try{allDocs=await sbGet('documents');render();}
  catch(e){grid.innerHTML=`<div class="empty-state"><div class="empty-icon">⚠️</div><p>문서 로딩 실패: ${e.message}</p></div>`;}
}

function render(){
  const list=curCat===0?allDocs:allDocs.filter(d=>d.category==curCat);
  const grid=document.getElementById('docGrid');
  updateBadges();
  if(!list.length){grid.innerHTML=`<div class="empty-state"><div class="empty-icon">📂</div><p>등록된 문서가 없습니다.</p></div>`;return;}
  const isAdmin=curUser&&curUser.role==='admin';
  grid.innerHTML=list.map(doc=>{
    const date=doc.created_at?doc.created_at.slice(0,10).replace(/-/g,'.'):'';
    return `<div class="doc-card">
      <div class="doc-icon">📄</div>
      <div class="doc-name">${esc(doc.title)}</div>
      <div class="doc-meta">${CATS[doc.category]?.name||''} · ${date} · ${esc(doc.uploader||'')}</div>
      <div class="doc-actions">
        <button class="btn-view" onclick="viewPdf('${esc(doc.file_path)}','${esc(doc.title)}')">👁 보기</button>
        <button class="btn-dl" onclick="dlPdf('${esc(doc.file_path)}','${esc(doc.filename||doc.title)}')">⬇ 다운로드</button>
        ${isAdmin?`<button class="btn-del" onclick="delDoc('${doc.id}','${esc(doc.file_path)}')">🗑</button>`:''}
      </div>
    </div>`;
  }).join('');
}

function setFile(f){
  if(!f)return;
  if(f.type!=='application/pdf'){alert('PDF 파일만 업로드 가능합니다.');return;}
  selFile=f;
  document.getElementById('fileSelected').textContent='✅ '+f.name+' ('+Math.round(f.size/1024)+'KB)';
}

async function doUpload(){
  const title=document.getElementById('upTitle').value.trim();
  const cat=parseInt(document.getElementById('upCat').value);
  if(!title){alert('문서 제목을 입력해주세요.');return;}
  if(!selFile){alert('PDF 파일을 선택해주세요.');return;}
  const pb=document.getElementById('progressBar');
  const pf=document.getElementById('progressFill');
  const pt=document.getElementById('progressText');
  const sb=document.getElementById('submitBtn');
  const cb=document.getElementById('cancelBtn');
  sb.disabled=true;sb.textContent='업로드 중...';cb.disabled=true;
  pb.style.display='block';pt.style.display='block';
  pf.style.width='20%';pt.textContent='파일 업로드 중...';
  try{
    const path=`psm_${cat}_${Date.now()}.pdf`;
    const upRes=await fetch(`${SB_URL}/storage/v1/object/${BUCKET}/${path}`,{
      method:'POST',
      headers:{'apikey':SB_KEY,'Authorization':'Bearer '+SB_KEY,'Content-Type':'application/pdf','x-upsert':'true'},
      body:selFile
    });
    if(!upRes.ok){const t=await upRes.text();throw new Error(`Storage 오류 ${upRes.status}: ${t}`);}
    pf.style.width='70%';pt.textContent='문서 정보 저장 중...';
    const dbRes=await fetch(`${SB_URL}/rest/v1/documents`,{
      method:'POST',
      headers:{...HEADERS,'Prefer':'return=representation'},
      body:JSON.stringify({title,category:cat,filename:selFile.name,file_path:path,uploader:curUser.name})
    });
    if(!dbRes.ok){const t=await dbRes.text();throw new Error(`DB 오류 ${dbRes.status}: ${t}`);}
    pf.style.width='100%';pt.textContent='완료!';
    await loadDocs();
    setTimeout(()=>{closeUpload();toast('업로드 완료!');},400);
  }catch(e){
    toast('업로드 실패: '+e.message,'error');
    sb.disabled=false;sb.textContent='업로드';cb.disabled=false;
    pb.style.display='none';pt.style.display='none';
  }
}

function viewPdf(path,title){
  const url=sbFileUrl(path);
  document.getElementById('pdfTitle').textContent=title;
  const viewer=`https://docs.google.com/viewer?url=${encodeURIComponent(url)}&embedded=true`;
  document.getElementById('pdfBody').innerHTML=`<iframe src="${viewer}" allowfullscreen style="width:100%;height:100%;border:none;"></iframe>`;
  document.getElementById('pdfModal').classList.add('show');
}
function closePdf(){document.getElementById('pdfModal').classList.remove('show');document.getElementById('pdfBody').innerHTML='';}

async function dlPdf(path,filename){
  try{
    const url=sbFileUrl(path);
    const r=await fetch(url);
    if(!r.ok)throw new Error('파일을 가져올 수 없습니다.');
    const blob=await r.blob();
    const a=document.createElement('a');
    a.href=URL.createObjectURL(blob);
    a.download=filename||'document.pdf';
    document.body.appendChild(a);a.click();document.body.removeChild(a);
    setTimeout(()=>URL.revokeObjectURL(a.href),1000);
    toast('다운로드 시작!');
  }catch(e){toast('다운로드 실패: '+e.message,'error');}
}

async function delDoc(id,path){
  if(!confirm('이 문서를 삭제하시겠습니까?'))return;
  try{
    await sbDelete('documents',id);
    await fetch(`${SB_URL}/storage/v1/object/${BUCKET}/${path}`,{method:'DELETE',headers:{'apikey':SB_KEY,'Authorization':'Bearer '+SB_KEY}});
    allDocs=allDocs.filter(d=>d.id!==id);render();toast('삭제 완료!');
  }catch(e){toast('삭제 실패: '+e.message,'error');}
}
